15  /**
16   * Some general operations and contract for Mojo implementations that will interact
17   * with the system_* API methods. Manages the Thrift connection and delegates to 
18   * executeOperation on the implementation. Implementations must still implement
19   * the execute method of the parent Mojo.
20   * 
21   * @author zznate
22   */
23  public abstract class AbstractSchemaCassandraMojo extends AbstractCassandraMojo {
24                 
25      
26      protected abstract void parseArguments() throws IllegalArgumentException;
27      
28      protected abstract ThriftApiOperation buildOperation(); 
29      
30      /**
31       * Parses the arguments then calls 
32       */
33      @Override
34      public void execute() throws MojoExecutionException, MojoFailureException 
35      {
36          try 
37          {
38              parseArguments();
39          } catch (IllegalArgumentException iae) 
40          {
41              throw new MojoExecutionException(iae.getMessage());
42          }
43          Utils.executeThrift(buildOperation());
44      }
45

48  }
