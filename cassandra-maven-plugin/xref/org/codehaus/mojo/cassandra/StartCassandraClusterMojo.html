19  package org.codehaus.mojo.cassandra;
20  
21  import org.apache.maven.plugin.MojoExecutionException;
22  import org.apache.maven.plugin.MojoFailureException;
23  import org.cassandraunit.DataLoader;
24  import org.cassandraunit.dataset.FileDataSet;
25  import org.cassandraunit.dataset.ParseException;
26  
27  import java.io.File;
28  import java.io.IOException;
29  import java.math.BigInteger;
30  import java.util.Arrays;
31  
32  /**
33   * Starts a Cassandra instance in the background.
34   *
35   * @author stephenc
36   * @goal start-cluster
37   * @threadSafe
38   * @phase pre-integration-test
39   */
40  public class StartCassandraClusterMojo
41      extends AbstractCqlLoadMojo
42  {
43      /**
44       * How long to wait for Cassandra to be started before finishing the goal. A value of 0 will wait indefinately. A
45       * value of -1 will not wait at all.
46       *
47       * @parameter default-value="180"
48       */
49      protected int startWaitSeconds;
50  
51      /**
52       * When {@code true}, if this is a clean start then the load script will be applied automatically.
53       *
54       * @parameter expression="${cassandra.load.after.first.start}" default-value="true"
55       */
56      private boolean loadAfterFirstStart;
57  
58      /**
59       * The CassandraUnit dataSet to load.
60       *
61       * @parameter default-value="${basedir}/src/test/resources/dataSet.xml"
62       * @since 1.2.1-2
63       */
64      protected File cuDataSet;
65  
66      /**
67       * Whether to ignore errors when loading the script.
68       *
69       * @parameter expression="${cassandra.cu.load.failure.ignore}"
70       * @since 1.2.1-2
71       */
72      private boolean cuLoadFailureIgnore;
73  
74      /**
75       * When {@code true}, if this is a clean start then the CassandraUnit dataSet will be applied automatically.
76       *
77       * @parameter expression="${cassandra.cu.load.after.first.start}" default-value="true"
78       * @since 1.2.1-2
79       */
80      private boolean cuLoadAfterFirstStart;
81  
82      /**
83       * The number of nodes in the cluster.
84       *
85       * @parameter expression="${cassandra.cluster.size}" default-value="4"
86       */
87      private int clusterSize;
88  
89      /**
90       * {@inheritDoc}
91       */
92      public void execute()
93          throws MojoExecutionException, MojoFailureException
94      {
95          if ( skip )
96          {
97              getLog().info( "Skipping cassandra: cassandra.skip==true" );
98              return;
99          }
100         if ( clusterSize < 1 )
101         {
102             throw new MojoExecutionException(
103                 "Invalid cluster size of " + clusterSize + " specified. Must be at least 1" );
104         }
105         if ( clusterSize > 254 )
106         {
107             throw new MojoExecutionException(
108                 "Invalid cluster size of " + clusterSize + " specified. Must be less than 254" );
109         }
110         File[] cassandraDir = new File[clusterSize];
111         BigInteger[] initialToken = new BigInteger[clusterSize];
112         String[] listenAddress = new String[clusterSize];
113         boolean isClean = true;
114         for ( int node = 0; node < clusterSize; node++ )
115         {
116             listenAddress[node] = "127.0.0." + ( node + 1 );
117             initialToken[node] = BigInteger.valueOf( 2 ).pow( 127 ).multiply( BigInteger.valueOf( node ) ).divide(
118                 BigInteger.valueOf( clusterSize ) );
119             cassandraDir[node] =
120                 new File( this.cassandraDir.getParent(), this.cassandraDir.getName() + "-node" + ( node + 1 ) );
121             if ( isClean && cassandraDir[node].isDirectory() )
122             {
123                 getLog().debug( "Re-using existing Cassandra cluster in " + cassandraDir[node].getAbsolutePath() );
124                 isClean = false;
125             }
126         }
127         long timeStamp = System.currentTimeMillis();
128         if ( isClean )
129         {
130             getLog().debug( "First start of Cassandra cluster in " + Arrays.asList( cassandraDir ) );
131         }
132         try
133         {
134             for ( int node = 0; node < clusterSize; node++ )
135             {
136                 getLog().info( "Starting for Cassandra Node " + ( node + 1 ) + "..." );
137                 Utils.startCassandraServer( cassandraDir[node],
138                                             newServiceCommandLine( cassandraDir[node], listenAddress[node],
139                                                                    listenAddress[node], initialToken[node],
140                                                                    listenAddress, node == 0, node == 0 ? jmxPort : 0 ),
141                                             createEnvironmentVars(), getLog() );
142             }
143 
144             if ( startWaitSeconds >= 0 )
145             {
146                 for ( int node = 0; node < clusterSize; node++ )
147                 {
148                     getLog().info( "Waiting for Cassandra Node " + ( node + 1 ) + " to start..." );
149                     boolean started =
150                         Utils.waitUntilStarted( listenAddress[node], rpcPort, startWaitSeconds, getLog() );
151                     if ( !started )
152                     {
153                         Utils.stopCassandraServer( listenAddress[node], rpcPort, listenAddress[node], stopPort, stopKey,
154                                                    getLog() );
155                         throw new MojoFailureException( "Cassandra failed to start within " + startWaitSeconds + "s" );
156                     }
157                 }
158             }
159             if ( isClean && loadAfterFirstStart)
160             {
161                 execCqlFile();
162             }
163 
164             if ( isClean && cuLoadAfterFirstStart && cuDataSet != null && cuDataSet.isFile() )
165             {
166                 getLog().info( "Loading CassandraUnit dataSet " + cuDataSet + "..." );
167                 try
168                 {
169                     DataLoader dataLoader = new DataLoader( "cassandraUnitCluster", rpcAddress + ":" + rpcPort );
170                     dataLoader.load( new FileDataSet( cuDataSet.getAbsolutePath() ) );
171                 }
172                 catch ( ParseException e )
173                 {
174                     if ( cuLoadFailureIgnore )
175                     {
176                         getLog().error( e.getMessage() + ". Ignoring as cuLoadFailureIgnore is true" );
177                     }
178                     else
179                     {
180                         throw new MojoExecutionException( "Error while loading CassandraUnit dataSet", e );
181                     }
182                 }
183                 getLog().info( "Finished " + cuDataSet + "." );
184             }
185 
186             getLog().info(
187                 "Cassandra started in " + ( ( System.currentTimeMillis() - timeStamp ) / 100L ) / 10.0 + "s" );
188         }
189         catch ( IOException e )
190         {
191             throw new MojoExecutionException( e.getLocalizedMessage(), e );
192         }
193     }
194 }
